    private void paint(){
        
        if((jRadioButton1.isSelected())||((ksztalt.equals("OVAL"))&&(wczytane.equals(true))))
        {      
            gDC.setColor(color);
            gDC.drawOval(x-(xSize/2), y-(ySize/2), xSize, ySize);
            //ShapeType sType, int x, int y, Color color, int size
            ShapeInfo sInfo = new ShapeInfo(OVAL, x, y, color, xSize, ySize);
            
            sInfo.setX(x);
            sInfo.setY(y);
            sInfo.setsType(OVAL);
            sInfo.setxSize(xSize);
            sInfo.setySize(ySize);
            sInfo.setColor(color);
            history.add(sInfo);
            
        }      
        if(jRadioButton2.isSelected()||((ksztalt.equals("RECTANGLE"))&&(wczytane.equals(true)))){
            ShapeInfo sInfo = new ShapeInfo(RECTANGLE, x, y, color, xSize, ySize);
            
            sInfo.setX(x);
            sInfo.setY(y);
            sInfo.setsType(RECTANGLE);
            sInfo.setxSize(xSize);
            sInfo.setySize(ySize);
            sInfo.setColor(color);
            history.add(sInfo);
            
            gDC.setColor(color);   
            gDC.drawRect(x-(xSize/2), y-(ySize/2), xSize, ySize);           
        }
        if(jRadioButton3.isSelected()||((ksztalt.equals("HEXAGON"))&&(wczytane.equals(true)))){ 
            ShapeInfo sInfo = new ShapeInfo(HEXAGON, x, y, color, xSize, ySize);
            
            sInfo.setX(x);
            sInfo.setY(y);
            sInfo.setsType(HEXAGON);
            sInfo.setxSize(xSize);
            sInfo.setySize(ySize);
            sInfo.setColor(color);
            history.add(sInfo);
            
            Polygon polygon= new Polygon();
            for (int i = 0; i < 6; i++){
            polygon.addPoint((int) (x-(xSize/2) * Math.cos(i * 2 * Math.PI / 6)),
            (int) (y-(ySize/2) * Math.sin(i * 2 * Math.PI / 6)));}
            gDC.setColor(color);
            gDC.drawPolygon(polygon);     
        } 
        if(jRadioButton4.isSelected()||((ksztalt.equals("TRIANGLE"))&&(wczytane.equals(true)))){  
            ShapeInfo sInfo = new ShapeInfo(TRIANGLE, x, y, color, xSize, ySize);
            
            sInfo.setX(x);
            sInfo.setY(y);
            sInfo.setsType(TRIANGLE);
            sInfo.setxSize(xSize);
            sInfo.setySize(ySize);
            sInfo.setColor(color);
            history.add(sInfo);
            
            Polygon polygon1= new Polygon();
            for (int i = 0; i < 3; i++){
            polygon1.addPoint((int) (x-(xSize/2) * Math.cos(i * 2 * Math.PI / 3)),
            (int) (y-(ySize/2) * Math.sin(i * 2 * Math.PI / 3)));}
            gDC.setColor(color);
            gDC.drawPolygon(polygon1);     
        }
        
    }
